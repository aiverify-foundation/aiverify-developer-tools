import { useEffect, useState } from 'react';

The following example display the width and height of the parent container.

export const DisplayWidgetSize = ({ getContainerObserver }) => {
  const [ container, setContainer ] = useState({
    width: 0,
    height: 0,
  })

  useEffect(() => {
    if (getContainerObserver) {
      getContainerObserver((width, height) => {
        setContainer({
          width,
          height
        })
      })
    }
  },[getContainerObserver])

  return (
    <div style={{ width:"100%", backgroundColor:"olive", color:"white" }}>
      Container size: width {container.width}px, height {container.height}px
    </div>
  )
}

<DisplayWidgetSize getContainerObserver={props.getContainerObserver} />
